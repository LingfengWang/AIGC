### 项目 1: 使用预训练 SD 模型生成图像 (入门)

**目标**: 10 分钟内生成第一张 AI 图像

```python
# file: 01_text_to_image.py
import torch
from diffusers import StableDiffusionPipeline

# 1. 加载预训练模型
model_id = "runwayml/stable-diffusion-v1-5"
pipe = StableDiffusionPipeline.from_pretrained(
    model_id,
    torch_dtype=torch.float16,  # 使用半精度节省显存
    safety_checker=None  # 禁用安全检查 (可选)
)

# 2. 移动到 GPU
pipe = pipe.to("cuda")

# 3. 生成图像
prompt = "a cute cat sitting on a rainbow, digital art, high quality"
image = pipe(prompt, num_inference_steps=50, guidance_scale=7.5).images[0]

# 4. 保存
image.save("output/cat_rainbow.png")
print("✅ 图像已保存!")
```

**运行**:
```bash
python 01_text_to_image.py
```

**预期输出**: 一张猫咪坐在彩虹上的图像

### 项目 2: 理解扩散过程 (进阶)

**目标**: 可视化扩散过程的每一步

```python
# file: 02_visualize_diffusion.py
import torch
from diffusers import DDIMScheduler, UNet2DModel
from PIL import Image
import numpy as np

# 1. 加载模型
model_id = "google/ddpm-cat-256"
unet = UNet2DModel.from_pretrained(model_id)
scheduler = DDIMScheduler.from_pretrained(model_id)

unet = unet.to("cuda")

# 2. 准备噪声
batch_size = 1
image = torch.randn(
    (batch_size, unet.config.in_channels, unet.config.sample_size, unet.config.sample_size)
)
image = image.to("cuda")

# 3. 设置采样步数
scheduler.set_timesteps(50)

# 4. 逐步去噪并保存中间结果
for i, t in enumerate(scheduler.timesteps):
    with torch.no_grad():
        # 预测噪声
        noise_pred = unet(image, t).sample
        
        # 去噪一步
        image = scheduler.step(noise_pred, t, image).prev_sample
    
    # 每 5 步保存一次
    if i % 5 == 0:
        # 转换到 PIL 图像
        image_pil = ((image / 2 + 0.5).clamp(0, 1) * 255).to("cpu", dtype=torch.uint8).numpy()[0].transpose(1, 2, 0)
        Image.fromarray(image_pil).save(f"output/diffusion_step_{i:03d}.png")
        print(f"Step {i}: 已保存")

print("✅ 扩散过程可视化完成!")
```

**预期输出**: 10 张图像，展示从纯噪声到清晰图像的渐变过程

### 项目 4: 从零实现简易 DDPM (深入理解)

**目标**: 不用 Diffusers，自己实现 DDPM

```python
# file: 04_simple_ddpm.py
import torch
import torch.nn as nn
import numpy as np

# 1. 定义简单的 UNet (简化版)
class SimpleUNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.time_embed = nn.Sequential(
            nn.Linear(1, 128),
            nn.SiLU(),
            nn.Linear(128, 128)
        )
        self.down = nn.Sequential(
            nn.Conv2d(3, 64, 3, padding=1),
            nn.ReLU(),
            nn.Conv2d(64, 128, 3, padding=1),
            nn.ReLU(),
        )
        self.up = nn.Sequential(
            nn.Conv2d(128, 64, 3, padding=1),
            nn.ReLU(),
            nn.Conv2d(64, 3, 3, padding=1),
        )
    
    def forward(self, x, t):
        # 时间嵌入
        t_emb = self.time_embed(t.float().unsqueeze(-1)).unsqueeze(-1).unsqueeze(-1)
        # 下采样
        x = self.down(x) + t_emb
        # 上采样
        x = self.up(x)
        return x

# 2. 定义噪声调度
class NoiseScheduler:
    def __init__(self, T=1000, beta_start=1e-4, beta_end=0.02):
        self.T = T
        self.betas = torch.linspace(beta_start, beta_end, T)
        self.alphas = 1. - self.betas
        self.alphas_cumprod = torch.cumprod(self.alphas, dim=0)
    
    def add_noise(self, x0, noise, t):
        """一次性添加 t 步的噪声"""
        sqrt_alpha_cumprod = torch.sqrt(self.alphas_cumprod[t])
        sqrt_one_minus_alpha = torch.sqrt(1 - self.alphas_cumprod[t])
        return sqrt_alpha_cumprod * x0 + sqrt_one_minus_alpha * noise
    
    def get_loss(self, model, x0):
        """计算训练损失"""
        batch_size = x0.shape[0]
        # 随机采样时间步
        t = torch.randint(0, self.T, (batch_size,))
        # 随机噪声
        noise = torch.randn_like(x0)
        # 加噪
        xt = self.add_noise(x0, noise, t)
        # 预测噪声
        noise_pred = model(xt, t)
        # MSE 损失
        return torch.nn.functional.mse_loss(noise_pred, noise)

# 3. 训练
def train():
    model = SimpleUNet()
    scheduler = NoiseScheduler()
    optimizer = torch.optim.Adam(model.parameters(), lr=3e-4)
    
    # 准备数据 (这里用随机数据示例)
    train_data = torch.randn(1000, 3, 32, 32)
    
    for epoch in range(10):
        total_loss = 0
        for x0 in train_data:
            loss = scheduler.get_loss(model, x0.unsqueeze(0))
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            total_loss += loss.item()
        
        print(f"Epoch {epoch}: Loss = {total_loss / len(train_data):.4f}")
    
    torch.save(model.state_dict(), "output/simple_ddpm.pth")
    print("✅ 训练完成!")

# 4. 采样
@torch.no_grad()
def sample(model, scheduler, shape=(1, 3, 32, 32), steps=50):
    model.eval()
    x = torch.randn(shape)  # 从纯噪声开始
    
    for t in reversed(range(0, scheduler.T, scheduler.T // steps)):
        t_tensor = torch.tensor([t]).expand(shape[0])
        noise_pred = model(x, t_tensor)
        
        # 简单去噪 (实际应该用更复杂的采样器)
        alpha = scheduler.alphas_cumprod[t]
        x = (x - noise_pred * (1 - alpha) / torch.sqrt(1 - alpha)) / torch.sqrt(alpha)
        x = x.clamp(-1, 1)
    
    return x

if __name__ == "__main__":
    train()
    
    # 采样
    model = SimpleUNet()
    model.load_state_dict(torch.load("output/simple_ddpm.pth"))
    scheduler = NoiseScheduler()
    samples = sample(model, scheduler)
    print("✅ 采样完成!")
```

## ❓ 常见问题

### Q: 显存不够怎么办？
A: 使用半精度 (`float16`), 减小 batch size, 或使用模型量化

### Q: 生成质量不好？
A: 调整 prompt, 增加采样步数，尝试不同的模型

### Q: 训练太慢？
A: 使用 LoRA 等高效微调方法，或使用预训练模型
